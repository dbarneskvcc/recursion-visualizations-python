          # mergesort a[lo..hi] using auxiliary array aux[lo..hi]
          def _sort(self, iter, lo, hi):
              if hi <= lo:
                  return
              mid = lo + int((hi - lo) / 2)
              self._sort(iter, lo, mid)
              self._sort(iter, mid + 1, hi)
              self._merge(iter, lo, mid, hi)
          }
