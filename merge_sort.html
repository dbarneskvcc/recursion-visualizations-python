          public static void Sort(IComparable[] a) {
              IComparable[] aux = new IComparable[a.length];
              sort(a, aux, 0, a.length-1);
          }

          // mergesort a[lo..hi] using auxiliary array aux[lo..hi]
          private static void sort(
              IComparable[] a,
              IComparable[] aux,
              int lo,
              int hi
            ) {
              if (hi <= lo) return;
              int mid = lo + (hi - lo) / 2;
              sort(a, aux, lo, mid);
              sort(a, aux, mid + 1, hi);
              merge(a, aux, lo, mid, hi);
          }
