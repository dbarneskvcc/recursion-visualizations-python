          // Recursive move discs from one peg to another.
          private static void MoveDisc(int n, char src, char temp, char dest) {
              if (n == 1) {
                  Console.WriteLine(
                      "Move {0} -> {1}", src, dest
                  );
                  return;
              }
              MoveDisc(n - 1, src, dest, temp);
              MoveDisc(1, src, temp, dest);
              MoveDisc(n - 1, temp, src, dest);
          }
